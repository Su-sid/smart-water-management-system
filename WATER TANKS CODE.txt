To insert data into a MySQL database using Node.js, you can use the mysql module to connect to the database and perform the INSERT query.

First, you need to install the mysql module using npm:

Copy code
npm install mysql
Then, you can create a connection to the database and use the query() method to execute the INSERT query:

Copy code
const mysql = require("mysql");

const connection = mysql.createConnection({
    host: "localhost",
    user: "root",
    password: "",
    database: "test"
});

connection.connect(function(error) {
    if (error) {
        console.error("Error connecting: " + error.stack);
        return;
    }

    console.log("Connected as id " + connection.threadId);
});

const name = "John";
const age = 30;

const sql = `INSERT INTO users (name, age) VALUES (?, ?)`;
const values = [name, age];

connection.query(sql, values, function(error, results) {
    if (error) {
        console.error("Error inserting data: " + error.stack);
        return;
    }

    console.log("Data inserted successfully");
});

connection.end();
In the code above, the mysql.createConnection() method is used to create a connection to the database. Then, the connection.query() method is used to execute the INSERT query and pass the values to be inserted into the database as the second argument.

After inserting the data, it is important to close the connection to the database using the connection.end() method. This will release the connection and prevent memory leaks.
